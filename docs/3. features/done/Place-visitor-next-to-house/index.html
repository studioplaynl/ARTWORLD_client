  async function goHome(id) {
    if (typeof id === 'string') {
      SceneSwitcher.switchScene('DefaultUserHome', id);
    } else if ($ShowItemsBar) {
      SceneSwitcher.switchScene(
        'DefaultUserHome',
        ManageSession.userProfile.id,
      );
    }
  }
